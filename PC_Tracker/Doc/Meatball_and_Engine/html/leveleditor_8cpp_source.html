370 void cLevelEditor::EventHandler()
371 {

374  // First Let's poll for Mouse button HELD Down
375  // You can hold right click to delete tiles constantly
376  Uint8 ms = SDL_GetMouseState(NULL,NULL);
377  if (ms & SDL_BUTTON(SDL_BUTTON_RIGHT))
378  {
379  DeleteObject();

381 
382  if (ms & SDL_BUTTON(SDL_BUTTON_LEFT) && Mouse_command == MOUSE_COMMAND_SELECT_MULTISELECT_TILES)

384  MultiSelect->SetObjects();

387  UniversalEventHandler(&event);

389  switch ( event.type )

391  case SDL_QUIT:

400  break;
401 
402  }
403  case SDL_KEYDOWN:
404  {
405  // ESCAPE to ESCAPE mouse command or LEVEL MODE
406  if( event.key.keysym.sym == SDLK_ESCAPE )
407  {
408

410  mode = MODE_GAME;

426

431

436  }
437  // IF LCTRL IS HELD
438  else if (event.key.keysym.mod & KMOD_LCTRL)
